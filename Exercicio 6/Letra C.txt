CREATE DATABASE Letra_b;

CREATE TABLE CURSO(id_curso INT PRIMARY KEY, nome_curso VARCHAR(100) NOT NULL, carga_horaria SMALLINT);
CREATE TABLE ALUNO(id_aluno INT NOT NULL PRIMARY KEY, nome_aluno VARCHAR(100) NOT NULL, email_aluno VARCHAR(100), fone_contato CHAR(14));
CREATE TABLE TURMA(id_turma INT NOT NULL PRIMARY KEY,data_matricula DATE NOT NULL, id_aluno INT NOT NULL REFERENCES ALUNO(id_aluno), id_curso INT NOT NULL REFERENCES CURSO(id_curso));
